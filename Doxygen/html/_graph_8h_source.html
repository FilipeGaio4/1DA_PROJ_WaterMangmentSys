355/*
356 * Auxiliary function to remove an outgoing edge (with a given destination (d))
357 * from a vertex (this).
358 * Returns true if successful, and false if such edge does not exist.
359 */
360template<class T>
361bool Vertex<T>::removeEdge(T in) {
362 bool removedEdge = false;
363 auto it = adj.begin();
364 while (it != adj.end()) {
365 Edge<T> *edge = *it;
366 Vertex<T> *dest = edge->getDest();
367 if (dest->getInfo() == in) {
368 it = adj.erase(it);
369 deleteEdge(edge);
370 removedEdge = true; // allows for multiple edges to connect the same pair of vertices (multigraph)
371 } else {
372 it++;
373 }
374 }
375 return removedEdge;
376}

653/*
654 * Performs a depth-first search (dfs) traversal in a graph (this).
655 * Returns a vector with the contents of the vertices by dfs order.
656 */
657template<class T>
658std::vector<T> Graph<T>::dfs() const {
659 std::vector<T> res;

668/*
669 * Performs a depth-first search (dfs) in a graph (this) from the source node.
670 * Returns a vector with the contents of the vertices by dfs order.
671 */
672template<class T>
673std::vector<T> Graph<T>::dfs(const T &source) const {
674 std::vector<int> res;
675 // Get the source vertex
676 auto s = findVertex(source);
677 if (s == nullptr) {
678 return res;
679 }
680 // Set that no vertex has been visited yet
681 for (auto v: vertexSet) {
682 v->setVisited(false);
683 }
684 // Perform the actual DFS using recursion

690/*
691 * Auxiliary function that visits a vertex (v) and its adjacent, recursively.
692 * Updates a parameter with the list of visited node contents.

694template<class T>
695void Graph<T>::dfsVisit(Vertex<T> *v, std::vector<T> &res) const {

700 if (!w->isVisited()) {

706/****************** BFS ********************/
707/*
708 * Performs a breadth-first search (bfs) in a graph (this), starting
709 * from the vertex with the given source contents (source).
710 * Returns a vector with the contents of the vertices by bfs order.
711 */
712template<class T>
713std::vector<T> Graph<T>::bfs(const T &source) const {
714 std::vector<int> res;
715 // Get the source vertex
716 auto s = findVertex(source);
717 if (s == nullptr) {
718 return res;
719 }
720
721 // Set that no vertex has been visited yet
722 for (auto v: vertexSet) {
723 v->setVisited(false);
724 }
725
726 // Perform the actual BFS using a queue
727 std::queue<Vertex<T> *> q;
728 q.push(s);
729 s->setVisited(true);
730 while (!q.empty()) {
731 auto v = q.front();
732 q.pop();
733 res.push_back(v->getInfo());
734 for (auto &e: v->getAdj()) {
735 auto w = e->getDest();
736 if (!w->isVisited()) {
737 q.push(w);
738 w->setVisited(true);
739 }
740 }
741 }
742 return res;
743}

745/****************** isDAG ********************/
746/*
747 * Performs a depth-first search in a graph (this), to determine if the graph
748 * is acyclic (acyclic directed graph or DAG).
749 * During the search, a cycle is found if an edge connects to a vertex
750 * that is being processed in the stack of recursive calls (see theoretical classes).
751 * Returns true if the graph is acyclic, and false otherwise.
752 */
753
754template<class T>
755bool Graph<T>::isDAG() const {
756 for (auto v: vertexSet) {
757 v->setVisited(false);
758 v->setProcesssing(false);
759 }
760 for (auto v: vertexSet) {
761 if (!v->isVisited()) {
762 if (!dfsIsDAG(v)) return false;
763 }
764 }
765 return true;
766}
767
772template<class T>
773bool Graph<T>::dfsIsDAG(Vertex<T> *v) const {
774 v->setVisited(true);
775 v->setProcesssing(true);
776 for (auto e: v->getAdj()) {
777 auto w = e->getDest();
778 if (w->isProcessing()) return false;
779 if (!w->isVisited()) {
780 if (!dfsIsDAG(w)) return false;
781 }
782 }
783 v->setProcesssing(false);
784 return true;
785}

792/*
793 * Performs a topological sorting of the vertices of a graph (this).
794 * Returns a vector with the contents of the vertices by topological order.
795 * If the graph has cycles, returns an empty vector.

799template<class T>
800std::vector<T> Graph<T>::topsort() const {
801 std::vector<int> res;
802
803 for (auto v: vertexSet) {
804 v->setIndegree(0);
805 }
806 for (auto v: vertexSet) {
807 for (auto e: v->getAdj()) {
808 unsigned int indegree = e->getDest()->getIndegree();
809 e->getDest()->setIndegree(indegree + 1);
810 }
811 }
812
813 std::queue<Vertex<T> *> q;
814 for (auto v: vertexSet) {
815 if (v->getIndegree() == 0) {
816 q.push(v);
817 }
818 }
819
820 while (!q.empty()) {
821 Vertex<T> *v = q.front();
822 q.pop();
823 res.push_back(v->getInfo());
824 for (auto e: v->getAdj()) {
825 auto w = e->getDest();
826 w->setIndegree(w->getIndegree() - 1);
827 if (w->getIndegree() == 0) {
828 q.push(w);
829 }
830 }
831 }
832
833 if (res.size() != vertexSet.size()) {
834 //std::cout << "Impossible topological ordering!" << std::endl;
835 res.clear();
836 return res;
837 }
838
839 return res;
